<?php

namespace App\Http\Controllers\Admin;

use App\SettingEmpresa;
use App\Empresa;
use Illuminate\Http\Request;
use App\Http\Controllers\Controller;

class SettingsEmpresaController extends Controller
{
    public function edit($id){
        $configuracionEmpresa = SettingEmpresa::findOrfail($id);
        $empresa = Empresa::findOrfail($configuracionEmpresa->empresa_id);

        return view('admin.settingsEmpresa.edit', compact('configuracionEmpresa', 'empresa'));
    }

    public function update(Request $request, $id){

        $datos = array(
            'empresa_id' => $request->get('empresa_id'),

            // AJUSTES GENERALES
            'ms_sc_colorPrimario' => $request->get('ms_sc_colorPrimario'),
            'ms_sc_colorFuentePrimario' => $request->get('ms_sc_colorFuentePrimario'),
            // FIN AJUSTES GENERALES

        );

        SettingEmpresa::whereId($id)->update($datos);
        return redirect()->route('admin.empresas.show', ['id' => $datos['empresa_id']])->with('flash', 'Los ajustes del minisitio han sido actualizados con exito.');
    }
}
